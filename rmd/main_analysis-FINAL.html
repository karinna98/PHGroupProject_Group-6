Function for Descriptive Statistics

A single function labeled “my_summary_func” to display the results of the stat summaries in B.1

#creating a function to summarize the statistical summaries of the continuous variables of interest  

my_summary_func <- function(x) { #defines a function
  list(   
    Mean = mean(x),#mean
    SD = sd (x), #standard deviation
    Variance = var(x), #vairance
    Five.Num.Summary = summary(x) #five summary
  )
}

Function to Perform Analysis

A single R function to perform analysis of graphs (QualityCare, RaceEthn5, Age, BMI)

# Box plot for categorical variables
my_visual_func_I <- function(data, variable) {
    ggplot(data, aes(x = factor(.data[[variable]]))) +
        geom_bar(fill = "skyblue") +
        labs(title = paste("Distribution of", variable), x = variable, y = "Count") +
        theme_minimal()
}
# Histogram for continuous variables
my_visual_func_II <- function(data, variable) {
    ggplot(data, aes(x = .data[[variable]])) +
        geom_histogram(fill = "orange", color = "black", bins = 30) +
        labs(title = paste("Distribution of", variable), x = variable, y = "Frequency") + 
        theme_minimal()
}
# Examples
my_visual_func_I(hints.data, "QualityCare")

my_visual_func_I(hints.data, "RaceEthn5")

my_visual_func_II(hints.data, "Age")

my_visual_func_II(hints.data, "BMI")
